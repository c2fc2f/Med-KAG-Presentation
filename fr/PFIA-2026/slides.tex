\begin{frame}[c]{Native RAG en médecine}
  \large
  \begin{itemize}
    \item<1->
      \textbf{MedRAG
      (\cite{xiong2024benchmarkingretrievalaugmentedgenerationmedicine})~:}
      Précision de 39.6\% \to{} 70.6\% sur PubMedQA avec GPT-4
      \vspace{10px}
    \item<2->
      \textbf{\cite{hassan2025optimizingmedicalquestionansweringsystems}~:}
      Précision de 55.4\% \to{} 71.8\% sur PubMedQA avec Llama 2
  \end{itemize}
\end{frame}

\begin{frame}[c]{Défauts des Native RAG}
  \large
  \begin{itemize}
    \item<1-> \textbf{Contradictions~:} Possibilité de conflits entre les
      documents récupérés.
    \item<2-> \textbf{Redondance~:} Duplication de l'information dans l'index.
    \item<3-> \textbf{Perte de liens~:} Pas de conservation sémantique
      explicite entre les chunks.
  \end{itemize}
\end{frame}

\begin{frame}[c]{Solution \& Définition~: GraphRAG (KAG)}
  \centering
  \includegraphics[width=\textwidth]{graphrag.png}
\end{frame}

\begin{frame}[c]{Graph RAG en médecine}
  \large
  \begin{itemize}
    \item<1->
      \textbf{MedGraphRAG (\cite{wu-etal-2025-medical})~:}
      Précision de 75.3\% \to{} 83.3\% sur PubMedQA avec GPT-4
      \vspace{10px}
    \item<2->
      \textbf{MedSumGraph (\cite{KIM2026103311})~:}
      Précision de 61.40\% \to{} 84.20\% sur PubMedQA avec Llama 3 (70b)
  \end{itemize}
\end{frame}

\begin{frame}[c]{Notre proposition}
  \centering
  \huge Med-KAG~: Une KAG médicale avec récupération par embedding
\end{frame}

\begin{frame}[c]{L'UMLS en Graphe de Connaissance}
  \begin{columns}[c]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item<1-> \textbf{Extraction~:} Analyse des fichiers RRF
          (MRCONSO, MRREL) issus du métathésaurus UMLS\@.
        \item<2-> \textbf{Filtrage~:} Sélection des concepts pertinents (CUI)
          et des relations (REL/RELA).
        \item<3-> \textbf{Modélisation~:} Conversion vers une base de données
          orientée graphe (Neo4j).
        \item<4-> \textbf{Statistique~:} Plus de 13 millions de n\oe{}uds et
          50 millions de relations interconnectées.
      \end{itemize}
    \end{column}

FOR (c:UMLSConcept) ON (c.embedding)
OPTIONS {
  indexConfig: {
    `vector.dimensions`: 768,
    `vector.similarity_function`: 'cosine'
  }
};
  \end{minted}
  \pause[]
  \begin{minted}[ignorelexererrors=true]{cypher}
CALL db.index.vector.queryNodes(
  'umls_index', $top_k, $embedding
)
YIELD node
MATCH p = (node)-[*0..$neighborhood]-(neighbor)
RETURN p;
  \end{minted}

\section{L'évaluation de Med-KAG sur PubMedQA}

\begin{frame}[c]{PubMedQA}
  \textbf{Question~:} Spinal subdural hematoma: a sequela of a ruptured
  intracranial aneurysm?

  \vspace{10px}

  \textbf{Options~:}
  \vspace{-7px}
  \begin{itemize}
    \item \textbf{A} yes
    \item \textbf{B} no
    \item \textbf{C} maybe
  \end{itemize}

  \vfill

  \color{gray}{Note~: PubMedQA contient 500 questions, toutes avec trois
  réponses possibles.}
\end{frame}

\begin{frame}[c]{Différentes variations de Med-KAG}
  \begin{itemize}
    \item<1-> \textbf{RAG}~: Le modèle augmenté par notre module de
      récupération.
      \vspace{10px}
    \item<2-> \textbf{RAG-Cleaner}~: Le modèle n'est pas contraint par un
      format de réponse strict lors de la génération. Un modèle tiers
      intervient a posteriori pour formater la réponse.
      \vspace{10px}
    \item<3-> \textbf{RAG-Sum-Cleaner}~: Le sous-graphe récupéré est converti
      en langage naturel par un modèle intermédiaire.
  \end{itemize}
\end{frame}

\begin{frame}[c]{Pourquoi les performances sont-elles inférieures~?}
  \begin{itemize}
    \item<1-> \textbf{Graphe très dense~:} Malgré un top-k et un voisinage
      faibles, les sous-graphes récupérés restent volumineux.
      \vspace{10px}
    \item<2-> \textbf{Lost in the middle~:} Noyé dans un contexte trop long,
      le modèle de langue manque l'information pertinente.
      \vspace{10px}
    \item<3-> \textbf{UMLS insuffisant~:} À lui seul, il ne suffit pas à
      l'aide à la décision, ses relations sémantiques sont peu adaptées au
      raisonnement diagnostique.
      \vspace{10px}
    \item<4-> \textbf{Modèles de petite taille~:} Pénalisent la précision,
      mais nécessaires pour une exécution locale.
  \end{itemize}
\end{frame}

\begin{frame}[c]{Discussion et Conclusion}
  \begin{itemize}
    \item<1-> \textbf{L'optimisation technique~:} La transition vers une
      génération de requêtes Text-to-Cypher guidée par LLM\@.
      \vspace{10px}
    \item<2-> \textbf{L'extension de l'évaluation~:} La validation du système
      sur d'autres jeux de données biomédicaux et sur plusieurs familles de
      LLM open-source.
      \vspace{10px}
    \item<3-> \textbf{La mesure de la fiabilité~:} L'intégration de métriques
      explicitement conçues pour quantifier la réduction des hallucinations.
      \vspace{10px}
    \item<4-> \textbf{L'application clinique~:} la validation sur des données
      cliniques réelles issues d'entrepôts de données de santé.
  \end{itemize}
\end{frame}
